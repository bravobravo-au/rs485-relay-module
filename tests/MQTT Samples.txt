MQTT commands supported



mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"on"}'
mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"off"}'
mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"toggle"}'

#Turn on input 1 for 10 seconds
mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"on", "DelayActionTime":"{\"seconds\":10}", "DelayAction": "off"}'

#Turn on input 1 for 1 minute
mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"on", "DelayActionTime":"{\"minutes\":1}", "DelayAction": "off"}'

#Turn on input 1 for 1 hour
mosquitto_pub -t 'RS485-002/CMD/1/31' -m '{"Output":"on", "DelayActionTime":"{\"hours\":1}", "DelayAction": "off"}'



#view full status of module
mosquitto_pub -t 'RS485-002/CMD/STATUS' -m ''

#view status for only one module
mosquitto_pub -t 'RS485-002/CMD/STATUS' -m '{"modulesModbusAddress":1}'

#View status for inputs 30 and 31 for module 1
mosquitto_pub -t 'RS485-002/CMD/STATUS' -m '{"modulesModbusAddress":1,"inputAddresses":[30,31]}'

#View status for inputs and outputs 30 and 31 for module 1
mosquitto_pub -t 'RS485-002/CMD/STATUS' -m '{"modulesModbusAddress":1,"inputAddresses":[30,31],"outputAddresses":[30,31]}'
